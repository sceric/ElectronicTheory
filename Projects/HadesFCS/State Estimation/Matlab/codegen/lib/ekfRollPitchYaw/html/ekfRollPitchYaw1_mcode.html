1function [ roll, pitch, yaw ] = ekfRollPitchYaw( gyr, acc, psiMag, V, T )
  2
  3    persistent flag
  4    persistent g
  5    persistent x
  6    persistent P
  7    persistent Q
  8    persistent R
  9    persistent noiseGyr
 10    persistent noiseAcc
 11    persistent noiseMag
 12    persistent noiseBias
 13    
 14    if isempty(flag)
 15       
 16        flag = 1;
 17        
 18        g = 9.81;
 19        
 20        x = zeros(7, 1);
 21        x(1) = 1.0;
 22        
 23        P = 10.0 * eye(7);
 24        
 25        noiseGyr  = 0.003455;
 26        noiseAcc  = 0.015696;
 27        noiseMag  = 0.004242;
 28        
 29        noiseBias = 0.000100;
 30        
 31        Q = diag([3.0 * noiseGyr ^ 2, 3.0 * noiseGyr ^ 2, 3.0 * noiseGyr ^ 2, 3.0 * noiseGyr ^ 2, ...
 32                    noiseBias ^ 2, noiseBias ^ 2, noiseBias ^ 2]);
 33                
 34        R = diag([noiseAcc ^ 2, noiseAcc ^ 2, noiseAcc ^ 2, noiseMag ^ 2]);
 35        
 36    end
 37    
 38    p = gyr(1);
 39    q = gyr(2);
 40    r = gyr(3);
 41    
 42    % Extract states
 43    q0 = x(1); q1 = x(2); q2 = x(3); q3 = x(4);    
 44    bp = x(5); bq = x(6); br = x(7);
 45    
 46    % State transition function f(x,u)
 47    x(1) = x(1) + 0.5 * T * (q1*(bp - p) + q2*(bq - q) + q3*(br - r));
 48    x(2) = x(2) + 0.5 * T * (q3*(bq - q) - q0*(bp - p) - q2*(br - r));
 49    x(3) = x(3) + 0.5 * T * (q1*(br - r) - q0*(bq - q) - q3*(bp - p));
 50    x(4) = x(4) + 0.5 * T * (q2*(bp - p) - q1*(bq - q) - q0*(br - r));
 51    
 52    % Normalise quaternion and re-extract states
 53    inorm = 1.0 / sqrt(x(1) * x(1) + x(2) * x(2) + x(3) * x(3) + x(4) * x(4));
 54    x(1) = x(1) * inorm;
 55    x(2) = x(2) * inorm;
 56    x(3) = x(3) * inorm;
 57    x(4) = x(4) * inorm;
 58    
 59    q0 = x(1); q1 = x(2); q2 = x(3); q3 = x(4);
 60    
 61    % Jacobian of f(x,u)
 62    A = 0.5 * [0, bp - p, bq - q, br - r,  q1,  q2,  q3; ...
 63               p - bp,      0, r - br, bq - q, -q0,  q3, -q2; ...
 64               q - bq, br - r,      0, p - bp, -q3, -q0,  q1; ...
 65               r - br, q - bq, bp - p,      0,  q2, -q1, -q0; ...
 66               0,      0,      0,      0,   0,   0,   0; ...
 67               0,      0,      0,      0,   0,   0,   0; ...
 68               0,      0,      0,      0,   0,   0,   0];
 69           
 70    % Update error covariance matrix
 71    P = P + T * (A * P + P * A' + Q);
 72    
 73    % Output function h(x,u)
 74    h = [ g*(2*q0*q2 - 2*q1*q3); ...
 75         -V*(br - r) - g*(2*q0*q1 + 2*q2*q3); ...
 76          V*(bq - q) + g*(2*q1*q1 + 2*q2*q2 - 1); ...
 77          atan2(2*q0*q3 + 2*q1*q2, - 2*q2*q2 - 2*q3*q3 + 1)];
 78      
 79    % Jacobian of h(x,u)
 80    fact = 1.0 / ((2*q2^2 + 2*q3^2 - 1)^2 + (2*q0*q3 + 2*q1*q2)^2);
 81    C = [ 2*g*q2, -2*g*q3,  2*g*q0, -2*g*q1, 0, 0,  0; ...
 82         -2*g*q1, -2*g*q0, -2*g*q3, -2*g*q2, 0, 0, -V; ...
 83               0,  4*g*q1,  4*g*q2,       0, 0, V,  0; ...
 84        -(2*q3*(2*q2^2 + 2*q3^2 - 1)) * fact, -(2*q2*(2*q2^2 + 2*q3^2 - 1)) * fact, -(((2*q1)/(2*q2^2 + 2*q3^2 - 1) - (4*q2*(2*q0*q3 + 2*q1*q2))/(2*q2^2 + 2*q3^2 - 1)^2)*(2*q2^2 + 2*q3^2 - 1)^2)*fact, -(((2*q0)/(2*q2^2 + 2*q3^2 - 1) - (4*q3*(2*q0*q3 + 2*q1*q2))/(2*q2^2 + 2*q3^2 - 1)^2)*(2*q2^2 + 2*q3^2 - 1)^2)*fact, 0, 0,  0];
 85      
 86    % Kalman gain
 87    K = P * C' / (C * P * C' + R);
 88    
 89    % Update error covariance matrix
 90    P = (eye(7) - K * C) * P;
 91    
 92    % Update state estimate
 93    x = x + K * ([acc(1); acc(2); acc(3); psiMag] - h);
 94    
 95    % Normalise quaternion and re-extract states
 96    inorm = 1.0 / sqrt(x(1) * x(1) + x(2) * x(2) + x(3) * x(3) + x(4) * x(4));
 97    x(1) = x(1) * inorm;
 98    x(2) = x(2) * inorm;
 99    x(3) = x(3) * inorm;
100    x(4) = x(4) * inorm;
101    q0 = x(1); q1 = x(2); q2 = x(3); q3 = x(4);
102    
103    % Convert to Euler angles
104    roll  = atan2(2 * (q0 * q1 + q2 * q3), 1 - 2 * (q1 * q1 + q2 * q2));
105    pitch = asin(2 * (q0 * q2 - q3 * q1));
106    yaw   = atan2(2 * (q0 * q3 + q1 * q2), 1 - 2 * (q2 * q2 + q3 * q3));
107end
